cmake_minimum_required(VERSION 3.0...3.28)

project(
  WebGPU_Tools
  VERSION 1.0
  LANGUAGES CXX
)

set(DAWN_FETCH_DEPENDENCIES ON)

add_subdirectory(deps/glfw)
add_subdirectory(deps/webgpu)
add_subdirectory(deps/glm)
add_subdirectory(deps/stb)

# sdl3 test
option(VENDORED "Use vendored libraries" ON)
if(VENDORED)
    # This assumes you have added SDL as a submodule in vendored/SDL
    add_subdirectory(deps/SDL3-3.1.1 EXCLUDE_FROM_ALL)
else()
    # 1. Look for a SDL3 package,
    # 2. look for the SDL3-shared component, and
    # 3. fail if the shared component cannot be found.
    find_package(SDL3 REQUIRED CONFIG REQUIRED COMPONENTS SDL3-shared)
endif()


add_library(WebGPU_Tools STATIC
  src/webgpu_tools/utils/webgpu.cpp
)

set_target_properties(WebGPU_Tools PROPERTIES
  CXX_STANDARD 20
  CXX_STANDARD_REQUIRED ON
)

target_include_directories(
  WebGPU_Tools PUBLIC
  ${PROJECT_SOURCE_DIR}/src
)

target_compile_definitions(WebGPU_Tools PUBLIC
  GLM_FORCE_SWIZZLE
  GLM_FORCE_DEPTH_ZERO_TO_ONE
)

target_link_libraries(WebGPU_Tools PUBLIC
  SDL3::SDL3
  # glfw
  glm stb
  webgpu dawn_glfw dawncpp dawn_utils
)
